\begin{frame}{Tại sao không dùng Modulithic? (Req 1)}
\begin{columns}[T]
\column{0.48\textwidth}
\begin{block}{Lý do chọn Microservices}
\begin{itemize}
  \item \textbf{Polyglot language}: chat-service dùng Go (goroutines) cho WebSocket fan-out hiệu năng cao; business logic dùng Java Spring Boot
  \item \textbf{Independent scaling}: chat-service cần scale riêng khi traffic tăng, không kéo theo auth-service
  \item \textbf{Independent deploy}: cập nhật media-service không ảnh hưởng chat real-time
  \item \textbf{Fault isolation}: media-service lỗi không kéo sập toàn bộ hệ thống
\end{itemize}
\end{block}

\column{0.48\textwidth}
\begin{block}{Nếu là Modulithic\ldots}
\begin{itemize}
  \item Không thể mix Go + Java trong 1 process
  \item Scale toàn bộ khi chỉ cần scale 1 module
  \item Deploy toàn bộ khi chỉ thay đổi 1 module
  \item 1 memory leak / goroutine leak làm sập tất cả
\end{itemize}
\end{block}

\end{columns}
\end{frame}

\section{Nguyên lý thiết kế Microservices}
% ============================================================

\begin{frame}{Single Responsibility Principle (Req 3)}
\begin{block}{SRP --- Mỗi service 1 trách nhiệm rõ ràng}
\begin{itemize}
  \item \textbf{chat-service}: CHỈ quản lý WebSocket connections + fan-out messages $\rightarrow$ không write business DB, không có business logic
  \item \textbf{media-service}: CHỈ quản lý upload lifecycle (generate presigned URL $\rightarrow$ MinIO $\rightarrow$ finalize metadata)
  \item \textbf{notification-service}: CHỈ consume events + persist + serve notification list
  \item \textbf{auth-service}: CHỈ authentication (issue/validate token, session management)
\end{itemize}
\end{block}
\begin{alertblock}{Điểm tranh luận}
message-service đảm nhận 2 vai trò: (1) write model cho messages, (2) project inbox read-model (CQRS). Có thể tách thành inbox-service nếu scale lên.
\end{alertblock}
\end{frame}

\begin{frame}{High Cohesion \& Low Coupling (Req 3)}
\begin{block}{High Cohesion --- Low Coupling}
\begin{itemize}
  \item \textbf{High Cohesion}: dữ liệu và logic cùng domain nằm trong 1 service, không scatter
  \item \textbf{Low Coupling}: chỉ \textbf{2 synchronous dependencies} (gRPC):
    \begin{itemize}
      \item \small auth $\rightarrow$ user (registration)
      \item \small message $\rightarrow$ media (validate attachments)
    \end{itemize}
  \item Mọi giao tiếp còn lại: \textbf{async event bus}
  \item Downstream services không parse JWT --- trust header từ Kong
\end{itemize}
\end{block}
\begin{block}{Execution Pipeline}
Request $\rightarrow$ Kong (validate) $\rightarrow$ Service $\rightarrow$ DB / Event Bus\\
Không có cross-service synchronous chain dài hơn 2 hops
\end{block}
\end{frame}

\section{Giao tiếp giữa các dịch vụ}
% ============================================================

\begin{frame}{Sync vs Async --- Khi nào dùng gì? (Req 4)}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{Synchronous --- gRPC (2 trường hợp)}
\begin{itemize}
  \item \textbf{auth $\rightarrow$ user-service} (\texttt{CreateProfile}):
    \begin{itemize}
      \item \small Cần response ngay để rollback nếu profile creation fail
      \item \small Deadline 5 giây, circuit breaker Resilience4j
    \end{itemize}
  \item \textbf{message $\rightarrow$ media-service} (\texttt{ValidateAttachments}):
    \begin{itemize}
      \item \small Validate media IDs trước khi persist message
      \item \small Deadline 3 giây (hot path user-facing)
    \end{itemize}
  \item Dùng gRPC vì: type-safe contract (Protobuf), binary encoding nhanh hơn REST, không qua Kong (internal network)
\end{itemize}
\end{block}

\column{0.5\textwidth}
\begin{block}{Asynchronous --- Event Bus (Kafka / RabbitMQ)}
\begin{itemize}
  \item \textbf{message.created} $\rightarrow$ chat-service (WebSocket fan-out), notification-service, InboxProjector
  \item \textbf{group.*} (created/member-added/removed/updated) $\rightarrow$ message-service (project Conversation)
  \item \textbf{user.created} $\rightarrow$ user-service
  \item Dùng async vì: fan-out 1-to-many, producers không cần chờ consumers, failure isolation
  \item \textbf{Switchable}: \texttt{EVENT\_BUS=rabbitmq} hoặc \texttt{kafka} --- cùng interface \texttt{EventBus} (Factory pattern)
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\section{Thiết kế \& Quản lý cơ sở dữ liệu}
% ============================================================

\begin{frame}{DB per Service \& Consistency (Req 6)}
\begin{block}{DB per Service}
\begin{itemize}
  \item 6 PostgreSQL instances độc lập + Redis + MinIO
  \item Không service nào truy cập trực tiếp DB của service khác
  \item \textbf{Strong consistency}: trong 1 service (ACID transactions)
  \item \textbf{Eventual consistency}: cross-service qua event bus
    \begin{itemize}
      \item \small group member change $\rightarrow$ message-service projection có độ trễ nhỏ
      \item \small message.created $\rightarrow$ inbox view update không đồng thời
    \end{itemize}
\end{itemize}
\end{block}
\begin{block}{Consistency cần thiết ở đâu?}
\begin{itemize}
  \item \textbf{Strong}: auth (login/register), message persist (không mất tin nhắn)
  \item \textbf{Eventual}: inbox unread count, notification, presence status
  \item \textbf{Saga}: \textcolor{red}{KHÔNG áp dụng} --- chỉ có 2-step registration (gRPC + circuit breaker đủ)
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{CAP Theorem \& Saga Pattern (Req 6)}
\begin{block}{CAP Theorem}
\begin{itemize}
  \item PostgreSQL: \textbf{CP} (Consistency + Partition Tolerance)

\begin{frame}{Scenario 1: Circuit Breaker Activation (Req 16)}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{Scenario: media-service xuống $\rightarrow$ CB open}
\begin{enumerate}
  \item Kill media-service pod:\\
    \texttt{kubectl delete pod media-service-xxx}
  \item Gửi message có attachment qua API
  \item Observe: \textbf{circuit breaker OPEN} sau $>$50\% failures trong 10 calls (Resilience4j)
  \item Response: fallback method trả về error (không hang)
  \item Restart media-service pod (K8s tự restart)
  \item Observe: CB $\rightarrow$ HALF-OPEN $\rightarrow$ 3 probe requests $\rightarrow$ CLOSED
\end{enumerate}
\end{block}

\column{0.5\textwidth}
\begin{alertblock}{[PLACEHOLDER --- chụp hình/log]}
\centering
\textit{Logs của message-service:}\\
\textit{--- CB state: CLOSED $\rightarrow$ OPEN}\\
\textit{--- Fallback called}\\
\textit{--- CB state: OPEN $\rightarrow$ HALF-OPEN}\\
\textit{--- CB state: HALF-OPEN $\rightarrow$ CLOSED}\\[0.3cm]
\textit{Hoặc Grafana metric panel cho}\\
\textit{Resilience4j state transitions}
\end{alertblock}
\begin{block}{Kết quả kỳ vọng}
\begin{itemize}
  \item Hệ thống không crash khi media-service lỗi
  \item Fallback response trả về ngay (không chờ timeout)
  \item Tự recover khi media-service lên lại
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Scenario 2: Pod Crash \& Self-Healing (Req 16)}
\begin{columns}[T]
\column{0.5\textwidth}
\begin{block}{Scenario: chat-service pod crash}
\begin{enumerate}
  \item Client A và B đang chat real-time
  \item \texttt{kubectl delete pod chat-service-xxx -n zalord-dev}
  \item K8s phát hiện pod gone $\rightarrow$ tạo pod mới (Deployment replicas=1)
  \item Frontend WebSocket: \textbf{exponential backoff reconnect} (cap 15s, logic trong \texttt{websocket.ts})
  \item Pod mới Ready $\rightarrow$ clients reconnect thành công
\end{enumerate}
\end{block}
